	$(".uruchom-okienko").click(function(e){
		e.preventDefault();

		$("body").css("overflow", "hidden");

		var idOkienka = $(this).attr("data-okienkoId");
		$(idOkienka).fadeIn("fast");

			$(idOkienka).fadeOut("fast", function(){

		});
